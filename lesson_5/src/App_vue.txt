<template>
  <div>
    <h2>Hello, {{ name }}</h2>
    <input type="text" v-model="name" />
    <button v-on:click="modalOpen">Open</button>
    <div class="modal" v-if="isModalOpened">
      <button v-on:click="modalClose">Close</button>
      <div
        class="rectangl"
        v-bind:style="`width: ${rect.x}px; height: ${rect.y}px`"
      >
        {{ square }}
      </div>
      <input type="text" v-model="rect.x" />
      <input type="text" v-model="rect.y" />
      <p>
        <b>S: </b><span>{{ rect.x * rect.y }} square mm</span>
      </p>
    </div>
    <ul>
      <li v-bind:key="index" v-for="(item, index) of list">
        <b>{{ item.name }}</b> <i>{{ item.lname }}</i>
      </li>
    </ul>
  </div>
</template>

<script>
export default {
  data() {
    return {
      name: "John",
      isModalOpened: false,
      rect: { x: 100, y: 300 },
      list: [
        { name: "John", lname: "Doe" },
        { name: "Jane", lname: "Doe" },
        { name: "John", lname: "Smith" },
      ],
    };
  },
  computed: {
    square() {
      return this.rect.x * this.rect.y;
    },
  },
  methods: {
    modalOpen() {
      this.isModalOpened = true;
    },
    modalClose() {
      this.isModalOpened = false;
    },
    mounted() {
      console.log("Init");
      // здесь обращение к серверу, запрос данных и тд
    },
  },
};
</script>
